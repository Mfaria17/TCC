A amostra Circuito Lógico permite ao usuário fazer circuitos usando portas e fios, que são atualizados sempre que um Link é modificado e em intervalos por uma função setTimeout em loop.

A função updateStates chama uma função para atualizar cada nó de acordo com o tipo, que usa a cor dos links no nó para determinar a cor daqueles que saem dele. Vermelho significa zero ou falso; verde significa um ou verdadeiro. Clicar duas vezes em um nó de entrada alterna entre verdadeiro/falso.

Passe o mouse sobre um nó para ver sua categoria, exibida usando um Adorno compartilhado definido como dica de ferramenta. Uma paleta à esquerda do diagrama principal permite que o usuário arraste e solte novos nós. Esses nós podem então ser vinculados usando portas que são definidas nos vários modelos de nós. Cada porta de entrada pode ter apenas um link de entrada, enquanto as portas de saída podem ter muitos links de saída. Isso é controlado pela propriedade GraphObject.toMaxLinks .

    function  updateStates ( )  { 
      var oldskip = myDiagram . skipsUndoManager ; 
      meuDiagrama . skipsUndoManager =  true ; 
      // faça todos os nós de "entrada" primeiro 
      myDiagram . nós . cada ( nó  =>  { 
        if  ( nó . categoria ===  "entrada" )  { 
          doInput ( nó ) ;
        } 
      } ) ; 
      // agora podemos fazer todos os outros tipos de nós 
      myDiagram . nós . cada ( node  =>  { 
        switch  ( node . category )  { 
          case  "and" :  doAnd ( node ) ;  break ; 
          case  " or " :  doOr ( node ) ;  break ; 
          case  "xor" :  doXor ( node ) ;  break ;
          caso  "não" :  doNot ( nó ) ;  quebrar ; 
          case  "nand" :  doNand ( nó ) ;  quebrar ; 
          case  "nem" :  doNor ( nó ) ;  quebrar ; 
          case  "xnor" :  doXnor ( nó ) ;  quebrar ; 
          case  "saída" :  doOutput ( nó ) ;  quebrar ; 
          caso  "entrada" : quebrar ;   // doInput já chamado, acima 
        } 
      } ) ; 
      meuDiagrama . skipsUndoManager = oldskip ; 
    }
